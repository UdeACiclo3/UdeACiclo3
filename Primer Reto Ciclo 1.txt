            valorObjetivo = 56.0

            estadoNutricional = "B"
            valorObjetivo = 58.0
        else:
            estadoNutricional = "C"
            valorObjetivo = 63.0

    if estadoNutricional == "A":
        aporteNutricionalDia = 2.0 * aporteCarbohidratos + aporteProteina - 2.0 * aporteVerdura

        aporteNutricionalDia = 0.6 * aporteCarbohidratos + aporteProteina - 4.0 * aporteVerdura
    else:
        aporteNutricionalDia = 0.5 * aporteCarbohidratos + 0.7 * aporteProteina - 2.0 * aporteVerdura

    metaParaObjetivo = abs(peso - valorObjetivo)
    diasParaObjetivo = int(metaParaObjetivo / abs(aporteNutricionalDia)) + 1

        print(f'El estado nutricional del paciente es {estadoNutricional} y se requieren {round(abs(diasParaObjetivo))} dias de dieta para que alcance el peso maximo')
    else:
        print(f'El estado nutricional del paciente es {estadoNutricional} y se requieren {round(abs(diasParaObjetivo))} dias de dieta para que alcance un peso saludable')
